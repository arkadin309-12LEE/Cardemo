# 🏢 빌딩 현황 (Buildings Status)

>[!buildings] 이 도시의 모든 빌딩들 (안내데스크)

```dataview
TABLE WITHOUT ID
  file.link as "🏢 빌딩명",
  choice(빌딩상태, 빌딩상태, "계획중") as "상태",
  choice(완료율, 완료율 + "%", "0%") as "진행률",
  choice(우선순위, 우선순위, "중간") as "우선순위",
  dateformat(file.mtime, "MM-dd HH:mm") as "최근 활동"
WHERE contains(file.folder, this.file.folder) 
  AND file.name != this.file.name 
  AND 타입 = "안내데스크_빌딩허브"
SORT 빌딩상태 ASC, 우선순위 DESC, file.mtime DESC
LIMIT 15
```

**🏗️ 빌딩 통계:**
```dataviewjs
const currentFolder = dv.current().file.folder;
const buildings = dv.pages()
  .where(p => p.file.folder.startsWith(currentFolder) 
    && p.file.name !== dv.current().file.name
    && p.타입 === "안내데스크_빌딩허브");

const totalBuildings = buildings.length;
const activeBuildings = buildings.where(p => p.file.mtime > dv.date("today").minus({days: 7})).length;
const completedBuildings = buildings.where(p => p.빌딩상태 === "완료").length;
const inProgressBuildings = buildings.where(p => p.빌딩상태 === "진행중").length;
const plannedBuildings = buildings.where(p => p.빌딩상태 === "계획중" || !p.빌딩상태).length;

// 전체 진행률 계산 (완료율 평균)
const avgProgress = totalBuildings > 0 ? 
  Math.round(buildings.values.map(p => p.완료율 || 0).reduce((a, b) => a + b, 0) / totalBuildings) : 0;

dv.table(["📊 구분", "수량", "비율"], [
  ["🏢 총 빌딩 수", totalBuildings, "100%"],
  ["✅ 완료된 빌딩", completedBuildings, totalBuildings > 0 ? Math.round(completedBuildings/totalBuildings*100) + "%" : "0%"],
  ["🔄 진행중 빌딩", inProgressBuildings, totalBuildings > 0 ? Math.round(inProgressBuildings/totalBuildings*100) + "%" : "0%"],
  ["📋 계획중 빌딩", plannedBuildings, totalBuildings > 0 ? Math.round(plannedBuildings/totalBuildings*100) + "%" : "0%"],
  ["📈 전체 진행률", avgProgress + "%", "평균"]
]);
```
